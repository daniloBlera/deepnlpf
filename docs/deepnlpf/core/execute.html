    def run_r(self, script, *args):
        import rpy2.robjects as ro
        r = ro.r
        print(">>>>", script)
        r.source(script)
        return r.main(*args)

    def run_java(self, jar_file, *args):
        try:
            import subprocess
            return subprocess.check_output(['java', '-jar', jar_file, *args], shell=False)
        except Exception as err:
            print(err)
